Todos os exercícios foram feitos no SQLServer.

1)Buscar os nomes de todos os alunos que frequentam alguma turma do professor 'JOAO PEDRO'.
SELECT A.NOME FROM TURMA T
INNER JOIN ALUNO A ON T.ALUNO_ID = A.ID
INNER JOIN PROFESSOR P ON T.PROFESSOR_ID = P.ID
WHERE P.NOME = 'JOAO PEDRO';

2)Buscar os dias da semana que tenham aulas da disciplina 'MATEMATICA'.
SELECT DISTINCT T.DIA_DA_SEMANA FROM TURMA T
INNER JOIN DISCIPLINA D ON T.DISCIPLINA_ID = D.ID
WHERE D.NOME = 'MATEMATICA';

3)Buscar todos os alunos que frequentem aulas de 'MATEMATICA' e também 'FISICA'.
SELECT DISTINCT A.NOME FROM ALUNO A
INNER JOIN TURMA T ON T.ALUNO_ID = A.ID
INNER JOIN DISCIPLINA D ON T.DISCIPLINA_ID = D.ID
WHERE D.NOME IN('MATEMATICA','FISICA');

4)Buscar as disciplinas que não tenham nenhuma turma.
SELECT D.NOME FROM DISCIPLINA D
WHERE D.ID NOT IN (SELECT T.DISCIPLINA_ID FROM TURMA T)

5)Buscar os alunos que frequentem aulas de 'MATEMATICA' exceto os que frequentem 'QUIMICA'.
SELECT A.NOME FROM ALUNO A
INNER JOIN TURMA T ON T.ALUNO_ID = A.ID 
INNER JOIN DISCIPLINA D ON T.DISCIPLINA_ID = D.ID 
WHERE D.NOME = 'MATEMATICA' AND T.ALUNO_ID NOT IN (SELECT T.ALUNO_ID FROM TURMA T 
WHERE T.DISCIPLINA_ID = (SELECT TOP 1 D.ID FROM DISCIPLINA D WHERE D.NOME = 'QUIMICA'));
